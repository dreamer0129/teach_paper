bash
# 查找OpenBLAS安装路径（以RockyLinux默认路径为例）

bash
# 安装编译依赖

make -j8
sudo make install

# 设置环境变量
export OpenBLAS_DIR=/usr/local/openblas/lib/cmake/openblas
方法四：强制指定链接参数
bash
# 直接指定库和头文件路径
cmake .. \
